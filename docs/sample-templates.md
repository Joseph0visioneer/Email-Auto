## 4. n8n 구현용 템플릿 매핑 로직

```javascript
// 템플릿 선택 함수
function selectTemplate(grade, daysUntilConference, country) {
  let templateKey = '';
  
  // 시기별 기본 템플릿
  if (daysUntilConference >= 30) {
    templateKey = 'initial_notice';
  } else if (daysUntilConference >= 14) {
    templateKey = 'detailed_info';
  } else if (daysUntilConference >= 7) {
    templateKey = 'final_check';
  } else {
    templateKey = 'urgent_notice';
  }
  
  // 등급별 추가 정보
  templateKey += `_${grade.toLowerCase()}`;
  
  // 국가별 맞춤 정보
  if (['한국', 'Korea'].includes(country)) {
    templateKey += '_ko';
  } else if (['일본', 'Japan'].includes(country)) {
    templateKey += '_jp';
  } else if (['태국', 'Thailand', '싱가폴', 'Singapore', '베트남', 'Vietnam'].includes(country)) {
    templateKey += '_sea';
  }
  
  return templateKey;
}

// 개인화 변수 치환
function personalizeTemplate(template, userData) {
  return template
    .replace(/{{이름}}/g, userData.name)
    .replace(/{{소속}}/g, userData.affiliation)
    .replace(/{{국가}}/g, userData.country)
    .replace(/{{발표시간}}/g, userData.presentationTime)
    .replace(/{{Travel_Grant}}/g, userData.travelGrant);
}
```

이제 이 템플릿들을 n8n 워크플로우에 적용할 준비가 되었습니다!
